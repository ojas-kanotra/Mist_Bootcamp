JavaScript is a programming language that makes websites interactive and dynamic. While HTML creates the structure and CSS adds styling, JavaScript adds behavior - like responding to clicks, validating forms, creating animations, and updating content without refreshing the page. It's what makes modern websites feel alive and responsive to user actions.

ADDING JAVASCRIPT TO HTML
==========================
<script> alert("Hello!"); </script>             JavaScript code inside HTML file
<script src="script.js"></script>               Link to separate JavaScript file (recommended)
console.log("Hello World");                     Shows message in browser's developer console
alert("Hello World");                           Shows popup message box to user
confirm("Are you sure?");                       Shows yes/no dialog box
prompt("What's your name?");                    Shows input dialog box

COMPARING VALUES
================
x == y;                                         Checks if x and y are equal
x === y;                                        Checks if x and y are exactly equal (same type)
x != y;                                         Checks if x and y are not equal
x !== y;                                        Checks if x and y are not exactly equal
x > y;                                          Checks if x is greater than y
x < y;                                          Checks if x is less than y
x >= y;                                         Checks if x is greater than or equal to y
x <= y;                                         Checks if x is less than or equal to y

LOGICAL OPERATORS
=================
x && y;                                         Both x and y must be true
x || y;                                         Either x or y must be true
!x;                                             Opposite of x (if x is true, this is false)

FUNCTIONS (REUSABLE CODE)
==========================

    alert("Hello!");
}

    return "Hello " + name;
}
function add(a, b) {                            Function that adds two numbers
    return a + b;
}

greet("John");                                  Runs greet function with "John" as input
let result = add(5, 3);                         Runs add function and stores result (8)

LOOPS (REPEATING CODE)
=======================
for (let i = 0; i < 5; i++) {                   Repeats code 5 times (i goes 0,1,2,3,4)
    console.log("Count: " + i);
}
let fruits = ["apple", "banana", "orange"];     
for (let fruit of fruits) {                     Goes through each item in array
    console.log(fruit);
}
let count = 0;
while (count < 3) {                             Repeats while condition is true
    console.log("Count: " + count);
    count++;
}

CHANGING HTML WITH JAVASCRIPT
==============================
document.getElementById("myButton");            Finds HTML element with id="myButton"
document.querySelector(".myClass");             Finds first element with class="myClass"
document.querySelector("h1");                   Finds first h1 heading element
let title = document.getElementById("title");   Gets element and stores it in variable

title.style.color = "red";                      Changes text color to red
title.style.fontSize = "24px";                  Changes font size
element.classList.add("highlight");             Adds CSS class to element
element.classList.remove("hidden");             Removes CSS class from element

RESPONDING TO USER ACTIONS
============================
button.onclick = function() {                   When button is clicked, run this code
    alert("Button was clicked!");
};
button.addEventListener("click", function() {   Modern way to handle click events
    console.log("Button clicked!");
});
input.addEventListener("keyup", function() {    When user types in input field
    console.log("User is typing!");
});
form.addEventListener("submit", function() {    When form is submitted
    console.log("Form submitted!");
});

TIMING FUNCTIONS
